# MiniSNS GAS 実装ガイド

## 概要

このガイドでは、Google Apps Script (GAS) を使用した MiniSNS バックエンドの実装方法を詳しく説明します。

## 前提知識

- JavaScript (ES6+)
- Google Apps Script の基本
- Google Spreadsheet API
- RESTful API の概念

## プロジェクト構造

```
GAS/Script/
├── Config.js          # 定数定義・設定
├── Common.js          # 共通関数（ハッシュ・トークン・レスポンス生成）
├── Auth.js            # 認証関連（ログイン・登録・自動ログイン）
├── BordOperation.js   # 投稿操作（作成・取得・削除）
├── ReplyOperation.js  # 返信操作（作成・取得・削除）
└── Main.js            # エントリーポイント・ルーティング
```

#### hashPassword(password, salt)
```javascript
function hashPassword(password, salt) {
  const hash = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    password + salt
  );
  return hash.map(byte => {
    const unsigned = byte < 0 ? byte + 256 : byte;
    return unsigned.toString(16).padStart(2, '0');
  }).join('');
}
```

#### generateToken()
```javascript
function generateToken() {
  return Utilities.getUuid();
}
```

#### validateToken(token)
```javascript
function validateToken(token) {
  if (!token) return { valid: false, error: "トークンがありません" };

  const userSheet = getSheet(SHEET_NAMES.USERS);
  const data = userSheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row[8] === token && row[6] === 0) { // LoginToken, DeleteFlag
      const loginDate = new Date(row[7]); // LoginDate
      const now = new Date();
      const diffDays = (now - loginDate) / (1000 * 60 * 60 * 24);

      if (diffDays <= VALIDATION.TOKEN_EXPIRY_DAYS) {
        return { valid: true, userId: row[0] }; // UserID
      } else {
        return { valid: false, error: "トークンの有効期限が切れています" };
      }
    }
  }

  return { valid: false, error: "無効なトークンです" };
}
```

#### createResponse(status, message, data)
```javascript
function createResponse(status, message, data = null) {
  const response = {
    status: status,
    message: message,
    timestamp: new Date().toISOString()
  };

  if (data !== null) {
    response.data = data;
  }

  return ContentService
    .createTextOutput(JSON.stringify(response))
    .setMimeType(ContentService.MimeType.JSON);
}
```

### handleRegistUser(params)
```javascript
function handleRegistUser(params) {
  const { username, email, password, passwordConfirm } = params;

  // バリデーション
  if (!username || !email || !password || !passwordConfirm) {
    return createResponse("error", "必須項目が入力されていません");
  }

  if (password !== passwordConfirm) {
    return createResponse("error", "パスワードが一致しません");
  }

  if (username.length > VALIDATION.MAX_USERNAME_LENGTH) {
    return createResponse("error", "ユーザー名が長すぎます");
  }

  // 重複チェック
  const userSheet = getSheet(SHEET_NAMES.USERS);
  const data = userSheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    if (data[i][1] === username || data[i][2] === email) { // UserName, Mail
      return createResponse("error", "ユーザー名またはメールアドレスが既に使用されています");
    }
  }

  // ユーザーID生成
  const userId = "usr_" + Date.now() + "_" + Math.random().toString(36).substr(2, 9);
  const hashedPassword = hashPassword(password, userId);
  const now = new Date();

  // データ追加
  const newRow = [
    userId,           // UserID
    username,         // UserName
    email,            // Mail
    now,              // CreatedAt
    "",               // Authority
    now,              // UpdateAt
    0,                // DeleteFlag
    now,              // LoginDate
    "",               // LoginToken (後で設定)
    hashedPassword,   // password
    "",               // ProfileTxt
    ""                // ProfileImageID
  ];

  userSheet.appendRow(newRow);

  return createResponse("ok", "ユーザー登録が完了しました", {
    userId: userId,
    username: username,
    email: email
  });
}
```

### handleLogin(params)
```javascript
function handleLogin(params) {
  const { username, password } = params;

  if (!username || !password) {
    return createResponse("error", "ユーザー名とパスワードを入力してください");
  }

  const userSheet = getSheet(SHEET_NAMES.USERS);
  const data = userSheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row[1] === username && row[6] === 0) { // UserName, DeleteFlag
      const hashedPassword = hashPassword(password, row[0]); // UserID as salt

      if (row[9] === hashedPassword) { // password
        // トークン生成・更新
        const token = generateToken();
        const now = new Date();

        userSheet.getRange(i + 1, 9).setValue(token);  // LoginToken
        userSheet.getRange(i + 1, 8).setValue(now);    // LoginDate
        userSheet.getRange(i + 1, 6).setValue(now);    // UpdateAt

        return createResponse("ok", "ログインに成功しました", {
          userId: row[0],      // UserID
          username: row[1],    // UserName
          email: row[2],       // Mail
          token: token,
          profileTxt: row[10]  // ProfileTxt
        });
      }
    }
  }

  return createResponse("error", "ユーザー名またはパスワードが間違っています");
}
```

### handleAutoLogin(params)
```javascript
function handleAutoLogin(params) {
  const { token } = params;

  const validation = validateToken(token);
  if (!validation.valid) {
    return createResponse("error", validation.error);
  }

  const userSheet = getSheet(SHEET_NAMES.USERS);
  const data = userSheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row[0] === validation.userId && row[6] === 0) { // UserID, DeleteFlag
      // トークン再生成（スライディングセッション）
      const newToken = generateToken();
      const now = new Date();

      userSheet.getRange(i + 1, 9).setValue(newToken); // LoginToken
      userSheet.getRange(i + 1, 8).setValue(now);      // LoginDate
      userSheet.getRange(i + 1, 6).setValue(now);      // UpdateAt

      return createResponse("ok", "自動ログインに成功しました", {
        userId: row[0],        // UserID
        username: row[1],      // UserName
        email: row[2],         // Mail
        token: newToken,
        profileTxt: row[10]    // ProfileTxt
      });
    }
  }

  return createResponse("error", "ユーザーが見つかりません");
}
```

```javascript
function handleCreatePost(params) {
  const { token, content } = params;

  const validation = validateToken(token);
  if (!validation.valid) {
    return createResponse("error", validation.error);
  }

  if (!content || content.length > VALIDATION.MAX_POST_LENGTH) {
    return createResponse("error", "投稿内容が無効です");
  }

  const postId = "pst_" + Date.now() + "_" + Math.random().toString(36).substr(2, 9);
  const now = new Date();

  const postSheet = getSheet(SHEET_NAMES.POSTS);
  const newRow = [
    postId,           // PostID
    validation.userId, // UserID
    content,          // PostTxt
    now,              // PostDate
    now,              // UpdateDate
    0                 // DeleteFlag
  ];

  postSheet.appendRow(newRow);

  return createResponse("ok", "投稿が作成されました", {
    postId: postId,
    content: content,
    timestamp: now.toISOString()
  });
}
```

### handleGetPosts(params)
```javascript
function handleGetPosts(params) {
  const limit = parseInt(params.limit) || 20;
  const offset = parseInt(params.offset) || 0;

  const postSheet = getSheet(SHEET_NAMES.POSTS);
  const userSheet = getSheet(SHEET_NAMES.USERS);

  const postData = postSheet.getDataRange().getValues();
  const userData = userSheet.getDataRange().getValues();

  // ユーザー情報マップ作成
  const userMap = {};
  for (let i = 1; i < userData.length; i++) {
    const row = userData[i];
    if (row[6] === 0) { // DeleteFlag
      userMap[row[0]] = { // UserID
        username: row[1],     // UserName
        profileImageId: row[11] // ProfileImageID
      };
    }
  }

  // 投稿データ取得（新しい順）
  const posts = [];
  for (let i = postData.length - 1; i >= 1; i--) {
    const row = postData[i];
    if (row[5] === 0 && userMap[row[1]]) { // DeleteFlag, UserID
      posts.push({
        postId: row[0],       // PostID
        userId: row[1],       // UserID
        username: userMap[row[1]].username,
        content: row[2],      // PostTxt
        timestamp: row[3].toISOString(), // PostDate
        profileImageId: userMap[row[1]].profileImageId
      });
    }
  }

  // ページネーション
  const paginatedPosts = posts.slice(offset, offset + limit);

  return createResponse("ok", "投稿を取得しました", {
    posts: paginatedPosts,
    total: posts.length,
    limit: limit,
    offset: offset
  });
}
```

### handleDeletePost(params)
```javascript
function handleDeletePost(params) {
  const { token, postId } = params;

  const validation = validateToken(token);
  if (!validation.valid) {
    return createResponse("error", validation.error);
  }

  const postSheet = getSheet(SHEET_NAMES.POSTS);
  const data = postSheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row[0] === postId && row[1] === validation.userId && row[5] === 0) {
      // PostID, UserID, DeleteFlag
      postSheet.getRange(i + 1, 6).setValue(1); // DeleteFlag = 1
      postSheet.getRange(i + 1, 5).setValue(new Date()); // UpdateDate

      return createResponse("ok", "投稿を削除しました");
    }
  }

  return createResponse("error", "投稿が見つからないか、権限がありません");
}
```

```javascript
function handleCreateReply(params) {
  const { token, postId, content } = params;

  const validation = validateToken(token);
  if (!validation.valid) {
    return createResponse("error", validation.error);
  }

  if (!content || content.length > VALIDATION.MAX_POST_LENGTH) {
    return createResponse("error", "返信内容が無効です");
  }

  // 投稿の存在確認
  const postSheet = getSheet(SHEET_NAMES.POSTS);
  const postData = postSheet.getDataRange().getValues();
  let postExists = false;

  for (let i = 1; i < postData.length; i++) {
    if (postData[i][0] === postId && postData[i][5] === 0) {
      // PostID, DeleteFlag
      postExists = true;
      break;
    }
  }

  if (!postExists) {
    return createResponse("error", "指定された投稿が見つかりません");
  }

  const replyId = "rpl_" + Date.now() + "_" + Math.random().toString(36).substr(2, 9);
  const now = new Date();

  const replySheet = getSheet(SHEET_NAMES.REPLIES);
  const newRow = [
    replyId,           // ReplyID
    postId,            // PostID
    validation.userId, // UserID
    content,           // ReplyTxt
    now,               // ReplyDate
    now,               // UpdateDate
    0                  // DeleteFlag
  ];

  replySheet.appendRow(newRow);

  return createResponse("ok", "返信が作成されました", {
    replyId: replyId,
    postId: postId,
    content: content,
    timestamp: now.toISOString()
  });
}
```

```javascript
function handleGetReplies(params) {

    return createResponse("error", "投稿IDが指定されていません");
  }

  const userSheet = getSheet(SHEET_NAMES.USERS);

  const replyData = replySheet.getDataRange().getValues();
  const userData = userSheet.getDataRange().getValues();

      userMap[row[0]] = { // UserID
        username: row[1],     // UserName
        profileImageId: row[11] // ProfileImageID
      };
    }
  }

  // 返信データ取得（古い順）
  const replies = [];
  for (let i = 1; i < replyData.length; i++) {
    const row = replyData[i];
    if (row[1] === postId && row[6] === 0 && userMap[row[2]]) {

        replyId: row[0],       // ReplyID
        postId: row[1],        // PostID
        userId: row[2],        // UserID
        username: userMap[row[2]].username,
        content: row[3],       // ReplyTxt
        timestamp: row[4].toISOString(), // ReplyDate
        profileImageId: userMap[row[2]].profileImageId
      });
    }
  }

  // ページネーション
  const paginatedReplies = replies.slice(offset, offset + limit);

  return createResponse("ok", "返信を取得しました", {
    replies: paginatedReplies,
    total: replies.length,
```

### doPost(e)
```javascript
function doPost(e) {
  try {
    const params = JSON.parse(e.postData.contents);
    const mode = params.mode;

    switch (mode) {
      case "login":
        return handleLogin(params);
      case "auto_login":
        return handleAutoLogin(params);
      case "registUser":
        return handleRegistUser(params);
      case "createPost":
        return handleCreatePost(params);
      case "deletePost":
        return handleDeletePost(params);
      case "createReply":
        return handleCreateReply(params);
      case "deleteReply":
        return handleDeleteReply(params);
      default:
        return createResponse("error", "無効なモードです");
    }
  } catch (error) {
    console.error("doPost error:", error);
    return createResponse("error", "サーバーエラーが発生しました");
  }
}
```

## 8. トラブルシューティング

### よくあるエラー

1. **"シートが見つかりません"**
   - `Config.js` のシート名がスプレッドシートのシート名と一致するか確認

2. **"権限がありません"**
   - GASの権限承認が完了しているか確認
   - スプレッドシートの共有設定を確認

3. **"トークンエラー"**
   - トークンの有効期限（5日）を確認
   - トークンが正しく保存されているか確認

4. **CORSエラー**
   - GASは自動的にCORS対応済み
   - フロントエンドのAPI URLが正しいか確認

### デバッグ方法

1. GASエディタの「表示」→「ログ」で実行ログを確認
2. スプレッドシートでデータが正しく保存されているか確認
3. ブラウザの開発者ツールでネットワークリクエストを確認
